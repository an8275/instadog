CREATE DATABASE `InstaDog` /*!40100 DEFAULT CHARACTER SET utf8 COLLATE utf8_bin */



table chien

CREATE TABLE `chien` (
 `id` int(10) NOT NULL AUTO_INCREMENT,
 `nom` text COLLATE utf8_bin NOT NULL,
 `surnom` varchar(16) COLLATE utf8_bin NOT NULL,
 `sexe` text COLLATE utf8_bin NOT NULL,
 `age` int(2) NOT NULL,
 `race` text COLLATE utf8_bin NOT NULL,
 `photo` varchar(128) COLLATE utf8_bin NOT NULL,
 `user_id` int(10) NOT NULL,
 PRIMARY KEY (`id`),
 KEY `proprio` (`user_id`),
 CONSTRAINT `proprio` FOREIGN KEY (`user_id`) REFERENCES `user` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin


table user

	
CREATE TABLE `user` (
 `id` int(10) NOT NULL AUTO_INCREMENT,
 `pseudo` varchar(32) COLLATE utf8_bin NOT NULL,
 `email` varchar(32) COLLATE utf8_bin NOT NULL,
 `dateDerniereConnexion` date NOT NULL,
 `motDePasse` varchar(255) COLLATE utf8_bin NOT NULL,
 PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin

table commentaire

CREATE TABLE `commentaire` (
 `id` int(10) NOT NULL AUTO_INCREMENT,
 `contenu` varchar(256) COLLATE utf8_bin NOT NULL,
 `date` date NOT NULL,
 `user_id` int(10) NOT NULL,
 `article_id` int(10) NOT NULL,
 PRIMARY KEY (`id`),
 KEY `utili` (`user_id`),
 KEY `arti` (`article_id`),
 CONSTRAINT `arti` FOREIGN KEY (`article_id`) REFERENCES `article` (`id`) ON DELETE CASCADE ON UPDATE CASCADE,
 CONSTRAINT `utili` FOREIGN KEY (`user_id`) REFERENCES `user` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin


table article


CREATE TABLE `article` (
 `id` int(10) NOT NULL AUTO_INCREMENT,
 `img` varchar(128) COLLATE utf8_bin NOT NULL,
 `contenu` varchar(256) COLLATE utf8_bin NOT NULL,
 `dateArticle` date NOT NULL,
 `chien_id` int(10) NOT NULL,
 PRIMARY KEY (`id`),
 KEY `dogo` (`chien_id`),
 CONSTRAINT `dogo` FOREIGN KEY (`chien_id`) REFERENCES `chien` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin
